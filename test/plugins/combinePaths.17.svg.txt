Merge paths with media queries if they don't affect dangerous properties.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 20 20">
  <style>
    path {stroke:blue}
    @media (width > 600px) { 
    path { stroke:red } 
    }
  </style>
  <path d="M 1 3 h 10"/>
  <path d="M 1 5 h 10"/>
</svg>

@@@

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 20 20">
    <style>
        path {stroke:blue}
    @media (width &gt; 600px) { 
    path { stroke:red } 
    }
    </style>
    <path d="M1 3h10M1 5h10"/>
</svg>
