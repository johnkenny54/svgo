Multiply adjacent scales.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 1000 1000">
    <rect x="0" y="10" width="5" height="8" transform="scale(2 3) scale(4 5)"/>
    <rect x="10" y="30" width="5" height="8" transform="scale(2 3) scale(4)"/>
    <rect x="20" y="50" width="5" height="8" transform="scale(2) scale(3 4)"/>
    <rect x="30" y="70" width="5" height="8" transform="scale(2) scale(3)"/>
</svg>
